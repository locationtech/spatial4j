18  package com.spatial4j.core.io.jts;
19  
20  import com.spatial4j.core.context.SpatialContextFactory;
21  import com.spatial4j.core.context.jts.JtsSpatialContext;
22  import com.spatial4j.core.io.PolyshapeReader;
23  import com.spatial4j.core.io.PolyshapeWriter;
24  import com.spatial4j.core.shape.Shape;
25  import com.spatial4j.core.shape.jts.JtsGeometry;
26  import com.spatial4j.core.shape.jts.JtsPoint;
27  import com.vividsolutions.jts.geom.*;
28  import com.vividsolutions.jts.geom.impl.PackedCoordinateSequenceFactory;
29  
30  import java.io.IOException;
31  import java.util.ArrayList;
32  import java.util.List;
33  
34  public class JtsPolyshapeReader extends PolyshapeReader {
35  
36    protected final JtsSpatialContext ctx;
37  
38    public JtsPolyshapeReader(JtsSpatialContext ctx, SpatialContextFactory factory) {
39      super(ctx, factory);
40      this.ctx = ctx;
41    }
42    
43  
44    @Override
45    protected Shape makeCollection(List<? extends Shape> shapes) {
46      Class<?> last = null;
47      List<Geometry> geoms = new ArrayList<>(shapes.size());
48      for(Shape s : shapes) {
49        if(last!=null && last!=s.getClass()) {
50          return super.makeCollection(shapes);
51        }
52        if(s instanceof JtsGeometry) {
53          geoms.add(((JtsGeometry)s).getGeom());
54        }
55        else if(s instanceof JtsPoint) {
56          geoms.add(((JtsPoint)s).getGeom());
57        }
58        else {
59          return super.makeCollection(shapes);
60        }
61        last = s.getClass();
62      }
63      Geometry result = ctx.getGeometryFactory().buildGeometry(geoms);
64      if(result.getClass().equals(GeometryCollection.class)) {
65        return super.makeCollection(shapes);
66      }
67      // *not* calling makeShapeFromGeometry() since the underlying geometries here have
68      //  already been converted to shapes via that method (or equivalent).
69      return ctx.makeShape(result);
70    }
71  
72    // --------------------------------------------------------------
73    // Read GeoJSON
74    // --------------------------------------------------------------
75  
76    protected CoordinateSequence coordseq(List<double[]> list) {
77      CoordinateSequence seq =
78          PackedCoordinateSequenceFactory.DOUBLE_FACTORY.create(list.size(), 2);
79  
80      for (int i = 0; i < list.size(); i++) {
81        double[] point = list.get(i);
82        seq.setOrdinate(i, 0, point[0]);
83        seq.setOrdinate(i, 1, point[1]);
84      }
85      return seq;
86    }
87    
88    @Override
89    protected Shape readPolygon(XReader reader) throws IOException {
90      GeometryFactory gf = ctx.getGeometryFactory();
91      List<double[]> outer = reader.readPoints();
92  
93      LinearRing shell = gf.createLinearRing(coordseq(outer));
94      LinearRing[] holes = null;
95      if(!reader.isDone() && reader.peek()==PolyshapeWriter.KEY_ARG_START) {
96        List<LinearRing> list = new ArrayList<LinearRing>();
97        while(reader.isEvent() && reader.peek()==PolyshapeWriter.KEY_ARG_START) {
98          reader.readKey(); // eat the event;
99          list.add(gf.createLinearRing(coordseq(reader.readPoints())));
100       }
101       holes = list.toArray(new LinearRing[list.size()]);
102     }
103     return ctx.makeShapeFromGeometry(gf.createPolygon(shell, holes));
104   }
105 }
