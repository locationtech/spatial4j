25  /**
26   * Factory for a {@link SpatialContext} based on configuration data.  Call
27   * {@link #makeSpatialContext(java.util.Map, ClassLoader)} to construct one via String name-value
28   * pairs. To construct one via code then create a factory instance, set the fields, then call
29   * {@link #newSpatialContext()}.
30   * <p>
31   * The following keys are looked up in the args map:
32   * <DL>
33   * <DT>spatialContextFactory</DT>
34   * <DD>org.locationtech.spatial4j.context.SpatialContext or
35   * org.locationtech.spatial4j.context.jts.JtsSpatialContext</DD>
36   * <DT>geo</DT>
37   * <DD>true (default)| false -- see {@link SpatialContext#isGeo()} </DD>
38   * <DT>shapeFactoryClass</DT>
39   * <DD>Java class of the {@link ShapeFactory}.</DD>
40   * <DT>distCalculator</DT>
41   * <DD>haversine | lawOfCosines | vincentySphere | cartesian | cartesian^2
42   * -- see {@link DistanceCalculator}</DD>
43   * <DT>worldBounds</DT>
44   * <DD>{@code ENVELOPE(xMin, xMax, yMax, yMin)} -- see {@link SpatialContext#getWorldBounds()}</DD>
45   * <DT>normWrapLongitude</DT>
46   * <DD>true | false (default) -- see {@link SpatialContext#isNormWrapLongitude()}</DD>
47   * <DT>readers</DT>
48   * <DD>Comma separated list of {@link org.locationtech.spatial4j.io.ShapeReader} class names</DD>
49   * <DT>writers</DT>
50   * <DD>Comma separated list of {@link org.locationtech.spatial4j.io.ShapeWriter} class names</DD>
51   * <DT>binaryCodecClass</DT>
52   * <DD>Java class of the {@link org.locationtech.spatial4j.io.BinaryCodec}</DD>
53   * </DL>
54   */
55  public class SpatialContextFactory {

79    /**
80     * Creates a new {@link SpatialContext} based on configuration in
81     * <code>args</code>.  See the class definition for what keys are looked up
82     * in it.
83     * The factory class is looked up via "spatialContextFactory" in args
84     * then falling back to a Java system property (with initial caps). If neither are specified
85     * then {@link SpatialContextFactory} is chosen.
86     *
87     * @param args Non-null map of name-value pairs.
88     * @param classLoader Optional, except when a class name is provided to an
89     *                    argument.
90     */
91    public static SpatialContext makeSpatialContext(Map<String,String> args, ClassLoader classLoader) {
92      if (classLoader == null)
93        classLoader = SpatialContextFactory.class.getClassLoader();
94      SpatialContextFactory instance;
95      String cname = args.get("spatialContextFactory");
96      if (cname == null)
97        cname = System.getProperty("SpatialContextFactory");
98      if (cname == null)
99        instance = new SpatialContextFactory();
100     else {
101       try {
102         Class c = classLoader.loadClass(cname);
103         instance = (SpatialContextFactory) c.newInstance();
104       } catch (Exception e) {
105         throw new RuntimeException(e);
106       }
107     }
108     instance.init(args, classLoader);
109     return instance.newSpatialContext();
110   }

112   protected void init(Map<String, String> args, ClassLoader classLoader) {
113     this.args = args;
114     this.classLoader = classLoader;
115 
116     initField("geo");
117 
118     initField("shapeFactoryClass");
119 
120     initCalculator();
121 
122     //init wktParser before worldBounds because WB needs to be parsed
123     initFormats();
124     initWorldBounds();
125 
126     initField("normWrapLongitude");
127 
128     initField("binaryCodecClass");
129   }

131   /** Gets {@code name} from args and populates a field by the same name with the value. */
132   protected void initField(String name) {
133     //  note: java.beans API is more verbose to use correctly (?) but would arguably be better
134     Field field;
135     try {
136       field = getClass().getField(name);
137     } catch (NoSuchFieldException e) {
138       throw new Error(e);
139     }
140     String str = args.get(name);
141     if (str != null) {
142       try {
143         Object o;
144         if (field.getType() == Boolean.TYPE) {
145           o = Boolean.valueOf(str);
146         } else if (field.getType() == Class.class) {
147           try {
148             o = classLoader.loadClass(str);
149           } catch (ClassNotFoundException e) {
150             throw new RuntimeException(e);
151           }
152         } else if (field.getType().isEnum()) {
153           o = Enum.valueOf(field.getType().asSubclass(Enum.class), str);
154         } else {
155           throw new Error("unsupported field type: "+field.getType());//not plausible at runtime unless developing
156         }
157         field.set(this, o);
158       } catch (IllegalAccessException e) {
159         throw new Error(e);
160       } catch (Exception e) {
161         throw new RuntimeException(
162             "Invalid value '"+str+"' on field "+name+" of type "+field.getType(), e);
163       }
164     }
165   }
166 
167   protected void initCalculator() {
168     String calcStr = args.get("distCalculator");
169     if (calcStr == null)
170       return;
171     if (calcStr.equalsIgnoreCase("haversine")) {
172       distCalc = new GeodesicSphereDistCalc.Haversine();
173     } else if (calcStr.equalsIgnoreCase("lawOfCosines")) {
174       distCalc = new GeodesicSphereDistCalc.LawOfCosines();
175     } else if (calcStr.equalsIgnoreCase("vincentySphere")) {
176       distCalc = new GeodesicSphereDistCalc.Vincenty();
177     } else if (calcStr.equalsIgnoreCase("cartesian")) {
178       distCalc = new CartesianDistCalc();
179     } else if (calcStr.equalsIgnoreCase("cartesian^2")) {
180       distCalc = new CartesianDistCalc(true);
181     } else {
182       throw new RuntimeException("Unknown calculator: "+calcStr);
183     }
184   }

186   /**
187    * Check args for 'readers' and 'writers'.  The value should be a comma separated list
188    * of class names.
189    * 
190    * The legacy parameter 'wktShapeParserClass' is also supported to add a specific WKT prarser
191    */
192   protected void initFormats() {
193     try {
194       String val = args.get("readers");
195       if (val != null) {
196         for (String name : val.split(",")) {
197           readers.add(Class.forName(name.trim(), false, classLoader).asSubclass(ShapeReader.class));
198         }
199       } else {//deprecated; a parameter from when this was a raw class
200         val = args.get("wktShapeParserClass");
201         if (val != null) {
202           //LoggerFactory.getLogger(getClass()).warn("Using deprecated argument: wktShapeParserClass={}", val);
203           readers.add(Class.forName(val.trim(), false, classLoader).asSubclass(ShapeReader.class));
204         }
205       }
206       val = args.get("writers");
207       if (val != null) {
208         for (String name : val.split(",")) {
209           writers.add(Class.forName(name.trim(), false, classLoader).asSubclass(ShapeWriter.class));
210         }
211       }
212     } catch (ClassNotFoundException ex) {
213       throw new RuntimeException("Unable to find format class", ex);
214     }
215   }

218   public SupportedFormats makeFormats(SpatialContext ctx) {
219     checkDefaultFormats();  // easy to override
220     
221     List<ShapeReader> read = new ArrayList<ShapeReader>(readers.size());
222     for (Class<? extends ShapeReader> clazz : readers) {
223       try {
224         read.add(makeClassInstance(clazz, ctx, this));
225       } catch (Exception ex) {
226         throw new RuntimeException(ex);
227       }
228     }
229     
230     List<ShapeWriter> write = new ArrayList<ShapeWriter>(writers.size());
231     for (Class<? extends ShapeWriter> clazz : writers) {
232       try {
233         write.add(makeClassInstance(clazz, ctx, this));
234       } catch (Exception ex) {
235         throw new RuntimeException(ex);
236       }
237     }
238     
239     return new SupportedFormats(
240         Collections.unmodifiableList(read), 
241         Collections.unmodifiableList(write));
242   }

244   /**
245    * If no formats were defined in the config, this will make sure GeoJSON and WKT are registered
246    */
247   protected void checkDefaultFormats() {
248     if (readers.isEmpty()) {
249       addReaderIfNoggitExists(GeoJSONReader.class);
250       readers.add(WKTReader.class);
251       readers.add(PolyshapeReader.class);
252       readers.add(LegacyShapeReader.class);
253     }
254     if (writers.isEmpty()) {
255       writers.add(GeoJSONWriter.class);
256       writers.add(WKTWriter.class);
257       writers.add(PolyshapeWriter.class);
258       writers.add(LegacyShapeWriter.class);
259     }
260   }

262   public void addReaderIfNoggitExists(Class<? extends ShapeReader> reader) {
263     try {
264       if (classLoader==null) {
265         Class.forName("org.noggit.JSONParser");
266       } else {
267         Class.forName("org.noggit.JSONParser", true, classLoader);
268       }
269       readers.add(reader);
270     } catch (ClassNotFoundException e) {
271       //LoggerFactory.getLogger(getClass()).warn("Unable to support GeoJSON Without Noggit");
272     }
273   }

298   @SuppressWarnings("unchecked")
299   private <T> T makeClassInstance(Class<? extends T> clazz, Object... ctorArgs) {
300     try {
301       Constructor<?> empty = null;
302 
303       //can't simply lookup constructor by arg type because might be subclass type
304       ctorLoop: for (Constructor<?> ctor : clazz.getConstructors()) {
305         Class[] parameterTypes = ctor.getParameterTypes();
306         if (parameterTypes.length == 0) {
307           empty = ctor; // the empty constructor;
308         }
309         if (parameterTypes.length != ctorArgs.length)
310           continue;
311         for (int i = 0; i < ctorArgs.length; i++) {
312           Object ctorArg = ctorArgs[i];
313           if (!parameterTypes[i].isAssignableFrom(ctorArg.getClass()))
314             continue ctorLoop;
315         }
316         return clazz.cast(ctor.newInstance(ctorArgs));
317       }
318 
319       // If an empty constructor exists, use that
320       if (empty != null) {
321         return clazz.cast(empty.newInstance());
322       }
323     } catch (Exception e) {
324       throw new RuntimeException(e);
325     }
326     throw new RuntimeException(clazz + " needs a constructor that takes: "
327         + Arrays.toString(ctorArgs));
328   }
