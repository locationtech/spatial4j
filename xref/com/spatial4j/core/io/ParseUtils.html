38    /**
39     * Given a string containing <i>dimension</i> values encoded in it, separated by commas, return a String array of length <i>dimension</i>
40     * containing the values.
41     *
42     * @param out         A preallocated array.  Must be size dimension.  If it is not it will be resized.
43     * @param externalVal The value to parse
44     * @param dimension   The expected number of values for the point
45     * @return An array of the values that make up the point (aka vector)
46     * @throws com.spatial4j.core.exception.InvalidShapeException if the dimension specified does not match the number of values in the externalValue.
47     */
48    public static String[] parsePoint(String[] out, String externalVal, int dimension) throws InvalidShapeException {
49      //TODO: Should we support sparse vectors?
50      if (out == null || out.length != dimension) out = new String[dimension];
51      int idx = externalVal.indexOf(',');
52      int end = idx;
53      int start = 0;
54      int i = 0;
55      if (idx == -1 && dimension == 1 && externalVal.length() > 0) {//we have a single point, dimension better be 1
56        out[0] = externalVal.trim();
57        i = 1;
58      } else if (idx > 0) {//if it is zero, that is an error
59        //Parse out a comma separated list of point values, as in: 73.5,89.2,7773.4
60        for (; i < dimension; i++) {
61          while (start < end && externalVal.charAt(start) == ' ') start++;
62          while (end > start && externalVal.charAt(end - 1) == ' ') end--;
63          if (start == end) {
64            break;
65          }
66          out[i] = externalVal.substring(start, end);
67          start = idx + 1;
68          end = externalVal.indexOf(',', start);
69          idx = end;
70          if (end == -1) {
71            end = externalVal.length();
72          }
73        }
74      }
75      if (i != dimension) {
76        throw new InvalidShapeException("incompatible dimension (" + dimension +
77                ") and values (" + externalVal + ").  Only " + i + " values specified");
78      }
79      return out;
80    }

82    /**
83     * Given a string containing <i>dimension</i> values encoded in it, separated by commas, return a double array of length <i>dimension</i>
84     * containing the values.
85     *
86     * @param out         A preallocated array.  Must be size dimension.  If it is not it will be resized.
87     * @param externalVal The value to parse
88     * @param dimension   The expected number of values for the point
89     * @return An array of the values that make up the point (aka vector)
90     * @throws com.spatial4j.core.exception.InvalidShapeException if the dimension specified does not match the number of values in the externalValue.
91     */
92    public static double[] parsePointDouble(double[] out, String externalVal, int dimension) throws InvalidShapeException{
93      if (out == null || out.length != dimension) out = new double[dimension];
94      int idx = externalVal.indexOf(',');
95      int end = idx;
96      int start = 0;
97      int i = 0;
98      if (idx == -1 && dimension == 1 && externalVal.length() > 0) {//we have a single point, dimension better be 1
99        out[0] = Double.parseDouble(externalVal.trim());
100       i = 1;
101     } else if (idx > 0) {//if it is zero, that is an error
102       //Parse out a comma separated list of point values, as in: 73.5,89.2,7773.4
103       for (; i < dimension; i++) {
104         //TODO: abstract common code with other parsePoint
105         while (start < end && externalVal.charAt(start) == ' ') start++;
106         while (end > start && externalVal.charAt(end - 1) == ' ') end--;
107         if (start == end) {
108           break;
109         }
110         out[i] = Double.parseDouble(externalVal.substring(start, end));
111         start = idx + 1;
112         end = externalVal.indexOf(',', start);
113         idx = end;
114         if (end == -1) {
115           end = externalVal.length();
116         }
117       }
118     }
119     if (i != dimension) {
120       throw new InvalidShapeException("incompatible dimension (" + dimension +
121               ") and values (" + externalVal + ").  Only " + i + " values specified");
122     }
123     return out;
124   }

126   /**
127    * Extract (by calling {@link #parsePoint(String[], String, int)} and validate the latitude and
128    * longitude contained in the String by making sure the latitude is between 90 & -90 and longitude
129    * is between -180 and 180.
130    * <p/>
131    * The latitude is assumed to be the first part of the string and the longitude the second part.
132    *
133    * @param latLonStr The string to parse.  Latitude is the first value, longitude is the second.
134    * @return The lat long
135    *
136    * @throws com.spatial4j.core.exception.InvalidShapeException if there was an error parsing
137    */
138   public static final double[] parseLatitudeLongitude(String latLonStr) throws InvalidShapeException {
139     return parseLatitudeLongitude(null, latLonStr);
140   }
141 
142   /**
143    * A variation of {@link #parseLatitudeLongitude(String)} that re-uses an output array.
144    * @see #parseLatitudeLongitude(String)
145    */
146   public static final double[] parseLatitudeLongitude(double[] outLatLon, String latLonStr) throws InvalidShapeException {
147     outLatLon = parsePointDouble(outLatLon, latLonStr, 2);
148 
149     if (outLatLon[0] < -90.0 || outLatLon[0] > 90.0) {
150       throw new InvalidShapeException(
151               "Invalid latitude: latitudes are range -90 to 90: provided lat: ["
152                       + outLatLon[0] + "]");
153     }
154 
155 
156     if (outLatLon[1] < -180.0 || outLatLon[1] > 180.0) {
157       throw new InvalidShapeException(
158               "Invalid longitude: longitudes are range -180 to 180: provided lon: ["
159                       + outLatLon[1] + "]");
160     }
161 
162     return outLatLon;
163   }
