18  package com.spatial4j.core.io.jts;
19  
20  import com.spatial4j.core.context.jts.JtsSpatialContext;
21  import com.spatial4j.core.context.jts.JtsSpatialContextFactory;
22  import com.spatial4j.core.exception.InvalidShapeException;
23  import com.spatial4j.core.io.BinaryCodec;
24  import com.spatial4j.core.shape.Shape;
25  import com.vividsolutions.jts.geom.Geometry;
26  import com.vividsolutions.jts.geom.PrecisionModel;
27  import com.vividsolutions.jts.io.InStream;
28  import com.vividsolutions.jts.io.OutStream;
29  import com.vividsolutions.jts.io.ParseException;
30  import com.vividsolutions.jts.io.WKBConstants;
31  import com.vividsolutions.jts.io.WKBReader;
32  import com.vividsolutions.jts.io.WKBWriter;
33  
34  import java.io.DataInput;
35  import java.io.DataOutput;
36  import java.io.IOException;
37  
38  /**
39   * Writes shapes in WKB, if it isn't otherwise supported by the superclass.
40   */
41  public class JtsBinaryCodec extends BinaryCodec {
42  
43    protected final boolean useFloat;//instead of double
44  
45    public JtsBinaryCodec(JtsSpatialContext ctx, JtsSpatialContextFactory factory) {
46      super(ctx, factory);
47      //note: ctx.geometryFactory hasn't been set yet
48      useFloat = (factory.precisionModel.getType() == PrecisionModel.FLOATING_SINGLE);
49    }
50  
51    @Override
52    protected double readDim(DataInput dataInput) throws IOException {
53      if (useFloat)
54        return dataInput.readFloat();
55      return super.readDim(dataInput);
56    }
57  
58    @Override
59    protected void writeDim(DataOutput dataOutput, double v) throws IOException {
60      if (useFloat)
61        dataOutput.writeFloat((float) v);
62      else
63        super.writeDim(dataOutput, v);
64    }
65  
66    @Override
67    protected byte typeForShape(Shape s) {
68      byte type = super.typeForShape(s);
69      if (type == 0) {
70        type = TYPE_GEOM;//handles everything
71      }
72      return type;
73    }
74  
75    @Override
76    protected Shape readShapeByTypeIfSupported(final DataInput dataInput, byte type) throws IOException {
77      if (type != TYPE_GEOM)
78        return super.readShapeByTypeIfSupported(dataInput, type);
79      return readJtsGeom(dataInput);
80    }
81  
82    @Override
83    protected boolean writeShapeByTypeIfSupported(DataOutput dataOutput, Shape s, byte type) throws IOException {
84      if (type != TYPE_GEOM)
85        return super.writeShapeByTypeIfSupported(dataOutput, s, type);
86      writeJtsGeom(dataOutput, s);
87      return true;
88    }
89  
90    public Shape readJtsGeom(final DataInput dataInput) throws IOException {
91      JtsSpatialContext ctx = (JtsSpatialContext)super.ctx;
92      WKBReader reader = new WKBReader(ctx.getGeometryFactory());
93      try {
94        InStream inStream = new InStream() {//a strange JTS abstraction
95          boolean first = true;
96          @Override
97          public void read(byte[] buf) throws IOException {
98            if (first) {//we don't write JTS's leading BOM so synthesize reading it
99              if (buf.length != 1)
100               throw new IllegalStateException("Expected initial read of one byte, not: " + buf.length);
101             buf[0] = WKBConstants.wkbXDR;//0
102             first = false;
103           } else {
104             //TODO for performance, specialize for common array lengths: 1, 4, 8
105             dataInput.readFully(buf);
106           }
107         }
108       };
109       Geometry geom = reader.read(inStream);
110       //false: don't check for dateline-180 cross or multi-polygon overlaps; this won't happen
111       // once it gets written, and we're reading it now
112       return ctx.makeShape(geom, false, false);
113     } catch (ParseException ex) {
114       throw new InvalidShapeException("error reading WKT", ex);
115     }
116   }
117 
118   public void writeJtsGeom(final DataOutput dataOutput, Shape s) throws IOException {
119     JtsSpatialContext ctx = (JtsSpatialContext)super.ctx;
120     Geometry geom = ctx.getGeometryFrom(s);//might even translate it
121     new WKBWriter().write(geom, new OutStream() {//a strange JTS abstraction
122       boolean first = true;
123       @Override
124       public void write(byte[] buf, int len) throws IOException {
125         if (first) {
126           first = false;
127           //skip byte order mark
128           if (len != 1 || buf[0] != WKBConstants.wkbXDR)//the default
129             throw new IllegalStateException("Unexpected WKB byte order mark");
130           return;
131         }
132         dataOutput.write(buf, 0, len);
133       }
134     });
135   }
136 }
