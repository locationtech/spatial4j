24  /**
25   * INTERNAL: A line between two points with a buffer distance extending in every direction. By
26   * contrast, an un-buffered line covers no area and as such is extremely unlikely to intersect with
27   * a point. BufferedLine isn't yet aware of geodesics (e.g. the dateline); it operates in Euclidean
28   * space.
29   */
30  public class BufferedLine extends BaseShape<SpatialContext> {
31  
32    private final Point pA, pB;
33    private final double buf;
34    private final Rectangle bbox;
35    /**
36     * the primary line; passes through pA & pB
37     */
38    private final InfBufLine linePrimary;
39    /**
40     * perpendicular to the primary line, centered between pA & pB
41     */
42    private final InfBufLine linePerp;
43  
44    /**
45     * Creates a buffered line from pA to pB. The buffer extends on both sides of
46     * the line, making the width 2x the buffer. The buffer extends out from
47     * pA & pB, making the line in effect 2x the buffer longer than pA to pB.
48     *
49     * @param pA  start point
50     * @param pB  end point
51     * @param buf the buffer distance in degrees
52     * @param ctx
53     */
54    public BufferedLine(Point pA, Point pB, double buf, SpatialContext ctx) {
55      super(ctx);
56      assert buf >= 0;//TODO support buf=0 via another class ?
57  
58      /**
59       * If true, buf should bump-out from the pA & pB, in effect
60       *                  extending the line a little.
61       */
62      final boolean bufExtend = true;//TODO support false and make this a
63      // parameter
64  
65      this.pA = pA;
66      this.pB = pB;
67      this.buf = buf;
68  
69      double deltaY = pB.getY() - pA.getY();
70      double deltaX = pB.getX() - pA.getX();
71  
72      PointImpl center = new PointImpl(pA.getX() + deltaX / 2,
73          pA.getY() + deltaY / 2, null);
74  
75      double perpExtent = bufExtend ? buf : 0;
76  
77      if (deltaX == 0 && deltaY == 0) {
78        linePrimary = new InfBufLine(0, center, buf);
79        linePerp = new InfBufLine(Double.POSITIVE_INFINITY, center, buf);
80      } else {
81        linePrimary = new InfBufLine(deltaY / deltaX, center, buf);
82        double length = Math.sqrt(deltaX * deltaX + deltaY * deltaY);
83        linePerp = new InfBufLine(-deltaX / deltaY, center,
84            length / 2 + perpExtent);
85      }
86  
87      double minY, maxY;
88      double minX, maxX;
89      if (deltaX == 0) { // vertical
90        if (pA.getY() <= pB.getY()) {
91          minY = pA.getY();
92          maxY = pB.getY();
93        } else {
94          minY = pB.getY();
95          maxY = pA.getY();
96        }
97        minX = pA.getX() - buf;
98        maxX = pA.getX() + buf;
99        minY = minY - perpExtent;
100       maxY = maxY + perpExtent;
101 
102     } else {
103       if (!bufExtend) {
104         throw new UnsupportedOperationException("TODO");
105         //solve for B & A (C=buf), one is buf-x, other is buf-y.
106       }
107 
108       //Given a right triangle of A, B, C sides, C (hypotenuse) ==
109       // buf, and A + B == the bounding box offset from pA & pB in x & y.
110       double bboxBuf = buf * (1 + Math.abs(linePrimary.getSlope()))
111           * linePrimary.getDistDenomInv();
112       assert bboxBuf >= buf && bboxBuf <= buf * 1.5;
113 
114       if (pA.getX() <= pB.getX()) {
115         minX = pA.getX() - bboxBuf;
116         maxX = pB.getX() + bboxBuf;
117       } else {
118         minX = pB.getX() - bboxBuf;
119         maxX = pA.getX() + bboxBuf;
120       }
121       if (pA.getY() <= pB.getY()) {
122         minY = pA.getY() - bboxBuf;
123         maxY = pB.getY() + bboxBuf;
124       } else {
125         minY = pB.getY() - bboxBuf;
126         maxY = pA.getY() + bboxBuf;
127       }
128 
129     }
130     Rectangle bounds = ctx.getWorldBounds();
131 
132     bbox = ctx.makeRectangle(
133         Math.max(bounds.getMinX(), minX),
134         Math.min(bounds.getMaxX(), maxX),
135         Math.max(bounds.getMinY(), minY),
136         Math.min(bounds.getMaxY(), maxY));
137   }
138 
139   @Override
140   public boolean isEmpty() {
141     return pA.isEmpty();
142   }
143 
144   @Override
145   public Shape getBuffered(double distance, SpatialContext ctx) {
146     return new BufferedLine(pA, pB, buf + distance, ctx);
147   }

149   /**
150    * Calls {@link DistanceUtils#calcLonDegreesAtLat(double, double)} given pA or pB's latitude;
151    * whichever is farthest. It's useful to expand a buffer of a line segment when used in
152    * a geospatial context to cover the desired area.
153    */
154   public static double expandBufForLongitudeSkew(Point pA, Point pB,
155                                                  double buf) {
156     double absA = Math.abs(pA.getY());
157     double absB = Math.abs(pB.getY());
158     double maxLat = Math.max(absA, absB);
159     double newBuf = DistanceUtils.calcLonDegreesAtLat(maxLat, buf);
160 //    if (newBuf + maxLat >= 90) {
161 //      //TODO substitute spherical cap ?
162 //    }
163     assert newBuf >= buf;
164     return newBuf;
165   }

167   @Override
168   public SpatialRelation relate(Shape other) {
169     if (other instanceof Point)
170       return contains((Point) other) ? CONTAINS : DISJOINT;
171     if (other instanceof Rectangle)
172       return relate((Rectangle) other);
173     throw new UnsupportedOperationException();
174   }
175 
176   public SpatialRelation relate(Rectangle r) {
177     //Check BBox for disjoint & within.
178     SpatialRelation bboxR = bbox.relate(r);
179     if (bboxR == DISJOINT || bboxR == WITHIN)
180       return bboxR;
181     //Either CONTAINS, INTERSECTS, or DISJOINT
182 
183     Point scratch = new PointImpl(0, 0, null);
184     Point prC = r.getCenter();
185     SpatialRelation result = linePrimary.relate(r, prC, scratch);
186     if (result == DISJOINT)
187       return DISJOINT;
188     SpatialRelation resultOpp = linePerp.relate(r, prC, scratch);
189     if (resultOpp == DISJOINT)
190       return DISJOINT;
191     if (result == resultOpp)//either CONTAINS or INTERSECTS
192       return result;
193     return INTERSECTS;
194   }
195 
196   public boolean contains(Point p) {
197     //TODO check bbox 1st?
198     return linePrimary.contains(p) && linePerp.contains(p);
199   }

251   @Override
252   public boolean equals(Object o) {
253     if (this == o) return true;
254     if (o == null || getClass() != o.getClass()) return false;
255 
256     BufferedLine that = (BufferedLine) o;
257 
258     if (Double.compare(that.buf, buf) != 0) return false;
259     if (!pA.equals(that.pA)) return false;
260     if (!pB.equals(that.pB)) return false;
261 
262     return true;
263   }
264 
265   @Override
266   public int hashCode() {
267     int result;
268     long temp;
269     result = pA.hashCode();
270     result = 31 * result + pB.hashCode();
271     temp = buf != +0.0d ? Double.doubleToLongBits(buf) : 0L;
272     result = 31 * result + (int) (temp ^ (temp >>> 32));
273     return result;
274   }
