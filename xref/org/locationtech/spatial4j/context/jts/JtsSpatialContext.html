22  /**
23   * Enhances the default {@link SpatialContext} with support for Polygons (and
24   * other geometries) using <a href="https://sourceforge.net/projects/jts-topo-suite/">JTS</a>.
25   * To the extent possible, our {@link JtsGeometry} adds some amount of geodetic support over
26   * vanilla JTS which only has a Euclidean (flat plane) model.
27   */
28  public class JtsSpatialContext extends SpatialContext {
29  
30    public static final JtsSpatialContext GEO;
31    static {
32      JtsSpatialContextFactory factory = new JtsSpatialContextFactory();
33      factory.geo = true;
34      GEO = new JtsSpatialContext(factory);
35    }
36  
37    /**
38     * Called by {@link org.locationtech.spatial4j.context.jts.JtsSpatialContextFactory#newSpatialContext()}.
39     */
40    public JtsSpatialContext(JtsSpatialContextFactory factory) {
41      super(factory);
42    }
43  
44    // TODO I expect to delete this eventually once the other deprecated methods in this class disappear
45    @Override
46    public JtsShapeFactory getShapeFactory() {
47      return (JtsShapeFactory) super.getShapeFactory();
48    }
49  
50    /**
51     * If geom might be a multi geometry of some kind, then might multiple
52     * component geometries overlap? Strict OGC says this is invalid but we
53     * can accept it by computing the union. Note: Our ShapeCollection mostly
54     * doesn't care but it has a method related to this
55     * {@link org.locationtech.spatial4j.shape.ShapeCollection#relateContainsShortCircuits()}.
56     */
57    @Deprecated
58    public boolean isAllowMultiOverlap() {
59      return getShapeFactory().isAllowMultiOverlap();
60    }
61  
62    /**
63     * Returns the rule used to handle geometry objects that have dateline crossing considerations.
64     */
65    @Deprecated
66    public DatelineRule getDatelineRule() {
67      return getShapeFactory().getDatelineRule();
68    }
69  
70    /**
71     * Returns the rule used to handle errors when creating a JTS {@link Geometry}, particularly after it has been
72     * read from one of the {@link ShapeReader}s.
73     */
74    @Deprecated
75    public ValidationRule getValidationRule() {
76      return getShapeFactory().getValidationRule();
77    }
78  
79    /**
80     * If JtsGeometry shapes should be automatically "prepared" (i.e. optimized) when read via from a {@link ShapeReader}.
81     *
82     * @see org.locationtech.spatial4j.shape.jts.JtsGeometry#index()
83     */
84    @Deprecated
85    public boolean isAutoIndex() {
86      return getShapeFactory().isAutoIndex();
87    }
88  
89    /**
90     * Gets a JTS {@link Geometry} for the given {@link Shape}. Some shapes hold a
91     * JTS geometry whereas new ones must be created for the rest.
92     * @param shape Not null
93     * @return Not null
94     */
95    @Deprecated
96    public Geometry getGeometryFrom(Shape shape) {
97      return getShapeFactory().getGeometryFrom(shape);
98    }
99  
100   /** Should {@link #makePoint(double, double)} return {@link JtsPoint}? */
101   @Deprecated
102   public boolean useJtsPoint() {
103     return getShapeFactory().useJtsPoint();
104   }
105 
106   /** Should {@link #makeLineString(java.util.List)} return {@link JtsGeometry}? */
107   @Deprecated
108   public boolean useJtsLineString() {
109     return getShapeFactory().useJtsLineString();
110   }
111 
112   /**
113    * INTERNAL Usually creates a JtsGeometry, potentially validating, repairing, and indexing ("preparing"). This method
114    * is intended for use by {@link ShapeReader} instances.
115    *
116    * If given a direct instance of {@link GeometryCollection} then it's contents will be
117    * recursively converted and then the resulting list will be passed to
118    * {@link SpatialContext#makeCollection(List)} and returned.
119    *
120    * If given a {@link com.vividsolutions.jts.geom.Point} then {@link SpatialContext#makePoint(double, double)}
121    * is called, which will return a {@link JtsPoint} if {@link JtsSpatialContext#useJtsPoint()}; otherwise
122    * a standard Spatial4j Point is returned.
123    *
124    * If given a {@link LineString} and if {@link JtsSpatialContext#useJtsLineString()} is true then
125    * then the geometry's parts are exposed to call {@link SpatialContext#makeLineString(List)}.
126    */
127   @Deprecated
128   public Shape makeShapeFromGeometry(Geometry geom) {
129     return getShapeFactory().makeShapeFromGeometry(geom);
130   }
131 
132   /**
133    * INTERNAL
134    * @see #makeShape(com.vividsolutions.jts.geom.Geometry)
135    *
136    * @param geom Non-null
137    * @param dateline180Check if both this is true and {@link #isGeo()}, then JtsGeometry will check
138    *                         for adjacent coordinates greater than 180 degrees longitude apart, and
139    *                         it will do tricks to make that line segment (and the shape as a whole)
140    *                         cross the dateline even though JTS doesn't have geodetic support.
141    * @param allowMultiOverlap See {@link #isAllowMultiOverlap()}.
142    */
143   @Deprecated
144   public JtsGeometry makeShape(Geometry geom, boolean dateline180Check, boolean allowMultiOverlap) {
145     return getShapeFactory().makeShape(geom, dateline180Check, allowMultiOverlap);
146   }
147 
148   /**
149    * INTERNAL: Creates a {@link Shape} from a JTS {@link Geometry}. Generally, this shouldn't be
150    * called when one of the other factory methods are available, such as for points. The caller
151    * needs to have done some verification/normalization of the coordinates by now, if any.  Also,
152    * note that direct instances of {@link GeometryCollection} isn't supported.
153    *
154    * Instead of calling this method, consider {@link JtsShapeFactory#makeShapeFromGeometry(Geometry)}
155    * which
156    */
157   @Deprecated
158   public JtsGeometry makeShape(Geometry geom) {
159     return getShapeFactory().makeShape(geom);
160   }
161 
162   @Deprecated
163   public GeometryFactory getGeometryFactory() {
164     return getShapeFactory().getGeometryFactory();
165   }
166 
167   @Override
168   public String toString() {
169     if (this.equals(GEO)) {
170       return GEO.getClass().getSimpleName()+".GEO";
171     } else {
172       return super.toString();
173     }
174   }
175 
176   /**
177    * INTERNAL: Returns a Rectangle of the JTS {@link Envelope} (bounding box) of the given {@code geom}.  This asserts
178    * that {@link Geometry#isRectangle()} is true.  This method reacts to the {@link DatelineRule} setting.
179    * @param geom non-null
180    * @return null equivalent Rectangle.
181    */
182   @Deprecated
183   public Rectangle makeRectFromRectangularPoly(Geometry geom) {
184     return getShapeFactory().makeRectFromRectangularPoly(geom);
185   }
186 }
