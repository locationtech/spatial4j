18  package com.spatial4j.core.shape.impl;
19  
20  import com.spatial4j.core.context.SpatialContext;
21  import com.spatial4j.core.shape.Point;
22  import com.spatial4j.core.shape.Rectangle;
23  import com.spatial4j.core.shape.Shape;
24  import com.spatial4j.core.shape.ShapeCollection;
25  import com.spatial4j.core.shape.SpatialRelation;
26  
27  import java.util.AbstractList;
28  import java.util.ArrayList;
29  import java.util.Collections;
30  import java.util.List;
31  
32  /**
33   * A BufferedLineString is a collection of {@link com.spatial4j.core.shape.impl.BufferedLine} shapes,
34   * resulting in what some call a "Track" or "Polyline" (ESRI terminology).
35   * The buffer can be 0.  Note that BufferedLine isn't yet aware of geodesics (e.g. the dateline).
36   */
37  public class BufferedLineString implements Shape {
38  
39    //TODO add some geospatial awareness like:
40    // segment that spans at the dateline (split it at DL?).
41  
42    private final ShapeCollection<BufferedLine> segments;
43    private final double buf;
44  
45    /**
46     * Needs at least 1 point, usually more than that.  If just one then it's
47     * internally treated like 2 points.
48     */
49    public BufferedLineString(List<Point> points, double buf, SpatialContext ctx) {
50      this(points, buf, false, ctx);
51    }
52  
53    /**
54     * @param points ordered control points. If empty then this shape is empty.
55     * @param buf Buffer >= 0
56     * @param expandBufForLongitudeSkew See {@link BufferedLine
57     * #expandBufForLongitudeSkew(com.spatial4j.core.shape.Point,
58     * com.spatial4j.core.shape.Point, double)}.
59     *                                  If true then the buffer for each segment
60     *                                  is computed.
61     * @param ctx
62     */
63    public BufferedLineString(List<Point> points, double buf, boolean expandBufForLongitudeSkew,
64                              SpatialContext ctx) {
65      this.buf = buf;
66  
67      if (points.isEmpty()) {
68        this.segments = ctx.makeCollection(Collections.<BufferedLine>emptyList());
69      } else {
70        List<BufferedLine> segments = new ArrayList<BufferedLine>(points.size() - 1);
71  
72        Point prevPoint = null;
73        for (Point point : points) {
74          if (prevPoint != null) {
75            double segBuf = buf;
76            if (expandBufForLongitudeSkew) {
77              //TODO this is faulty in that it over-buffers.  See Issue#60.
78              segBuf = BufferedLine.expandBufForLongitudeSkew(prevPoint, point, buf);
79            }
80            segments.add(new BufferedLine(prevPoint, point, segBuf, ctx));
81          }
82          prevPoint = point;
83        }
84        if (segments.isEmpty()) {//TODO throw exception instead?
85          segments.add(new BufferedLine(prevPoint, prevPoint, buf, ctx));
86        }
87        this.segments = ctx.makeCollection(segments);
88      }
89    }
90  
91    @Override
92    public boolean isEmpty() {
93      return segments.isEmpty();
94    }
95  
96    @Override
97    public Shape getBuffered(double distance, SpatialContext ctx) {
98      return ctx.makeBufferedLineString(getPoints(), buf + distance);
99    }
100 
101   public ShapeCollection<BufferedLine> getSegments() {
102     return segments;
103   }
104 
105   public double getBuf() {
106     return buf;
107   }
108 
109   @Override
110   public double getArea(SpatialContext ctx) {
111     return segments.getArea(ctx);
112   }
113 
114   @Override
115   public SpatialRelation relate(Shape other) {
116     return segments.relate(other);
117   }
118 
119   @Override
120   public boolean hasArea() {
121     return segments.hasArea();
122   }
123 
124   @Override
125   public Point getCenter() {
126     return segments.getCenter();
127   }
128 
129   @Override
130   public Rectangle getBoundingBox() {
131     return segments.getBoundingBox();
132   }
133 
134   @Override
135   public String toString() {
136     StringBuilder str = new StringBuilder(100);
137     str.append("BufferedLineString(buf=").append(buf).append(" pts=");
138     boolean first = true;
139     for (Point point : getPoints()) {
140       if (first) {
141         first = false;
142       } else {
143         str.append(", ");
144       }
145       str.append(point.getX()).append(' ').append(point.getY());
146     }
147     str.append(')');
148     return str.toString();
149   }
150 
151   public List<Point> getPoints() {
152     if (segments.isEmpty())
153       return Collections.emptyList();
154     final List<BufferedLine> lines = segments.getShapes();
155     return new AbstractList<Point>() {
156       @Override
157       public Point get(int index) {
158         if (index == 0)
159           return lines.get(0).getA();
160         return lines.get(index - 1).getB();
161       }
162 
163       @Override
164       public int size() {
165         return lines.size() + 1;
166       }
167     };
168   }
169 
170   @Override
171   public boolean equals(Object o) {
172     if (this == o) return true;
173     if (o == null || getClass() != o.getClass()) return false;
174 
175     BufferedLineString that = (BufferedLineString) o;
176 
177     if (Double.compare(that.buf, buf) != 0) return false;
178     if (!segments.equals(that.segments)) return false;
179 
180     return true;
181   }
182 
183   @Override
184   public int hashCode() {
185     int result;
186     long temp;
187     result = segments.hashCode();
188     temp = buf != +0.0d ? Double.doubleToLongBits(buf) : 0L;
189     result = 31 * result + (int) (temp ^ (temp >>> 32));
190     return result;
191   }
192 }
