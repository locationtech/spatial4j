20  import com.spatial4j.core.context.SpatialContext;
21  import com.spatial4j.core.shape.impl.Range;
22  
23  import java.util.AbstractList;
24  import java.util.ArrayList;
25  import java.util.Collection;
26  import java.util.List;
27  import java.util.RandomAccess;
28  
29  import static com.spatial4j.core.shape.SpatialRelation.CONTAINS;
30  import static com.spatial4j.core.shape.SpatialRelation.INTERSECTS;

32  /**
33   * A collection of Shape objects, analogous to an OGC GeometryCollection. The
34   * implementation demands a List (with random access) so that the order can be
35   * retained if an application requires it, although logically it's treated as an
36   * unordered Set, mostly.
37   * <p/>
38   * Ideally, {@link #relate(Shape)} should return the same result no matter what
39   * the shape order is, although the default implementation can be order
40   * dependent when the shapes overlap; see {@link #relateContainsShortCircuits()}.
41   *  To improve performance slightly, the caller could order the shapes by
42   * largest first so that relate() will have a greater chance of
43   * short-circuit'ing sooner.  As the Shape contract states; it may return
44   * intersects when the best answer is actually contains or within. If any shape
45   * intersects the provided shape then that is the answer.
46   * <p/>
47   * This implementation is not optimized for a large number of shapes; relate is
48   * O(N).  A more sophisticated implementation might do an R-Tree based on
49   * bbox'es, for example.
50   */
51  public class ShapeCollection<S extends Shape> extends AbstractList<S> implements Shape {
52  
53    protected final List<S> shapes;
54    protected final Rectangle bbox;
55  
56    /**
57     * WARNING: {@code shapes} is copied by reference.
58     * @param shapes Copied by reference! (make a defensive copy if caller modifies)
59     * @param ctx
60     */
61    public ShapeCollection(List<S> shapes, SpatialContext ctx) {
62      if (!(shapes instanceof RandomAccess))
63        throw new IllegalArgumentException("Shapes arg must implement RandomAccess: "+shapes.getClass());
64      this.shapes = shapes;
65      this.bbox = computeBoundingBox(shapes, ctx);
66    }
67  
68    protected Rectangle computeBoundingBox(Collection<? extends Shape> shapes, SpatialContext ctx) {
69      if (shapes.isEmpty())
70        return ctx.makeRectangle(Double.NaN, Double.NaN, Double.NaN, Double.NaN);
71      Range xRange = null;
72      double minY = Double.POSITIVE_INFINITY;
73      double maxY = Double.NEGATIVE_INFINITY;
74      for (Shape geom : shapes) {
75        Rectangle r = geom.getBoundingBox();
76  
77        Range xRange2 = Range.xRange(r, ctx);
78        if (xRange == null) {
79          xRange = xRange2;
80        } else {
81          xRange = xRange.expandTo(xRange2);
82        }
83        minY = Math.min(minY, r.getMinY());
84        maxY = Math.max(maxY, r.getMaxY());
85      }
86      return ctx.makeRectangle(xRange.getMin(), xRange.getMax(), minY, maxY);
87    }
88  
89    public List<S> getShapes() {
90      return shapes;
91    }
92  
93    @Override
94    public S get(int index) {
95      return shapes.get(index);
96    }
97  
98    @Override
99    public int size() {
100     return shapes.size();
101   }

103   @Override
104   public Rectangle getBoundingBox() {
105     return bbox;
106   }
107 
108   @Override
109   public Point getCenter() {
110     return bbox.getCenter();
111   }
112 
113   @Override
114   public boolean hasArea() {
115     for (Shape geom : shapes) {
116       if( geom.hasArea() ) {
117         return true;
118       }
119     }
120     return false;
121   }
122 
123   @Override
124   public ShapeCollection getBuffered(double distance, SpatialContext ctx) {
125     List<Shape> bufColl = new ArrayList<Shape>(size());
126     for (Shape shape : shapes) {
127       bufColl.add(shape.getBuffered(distance, ctx));
128     }
129     return ctx.makeCollection(bufColl);
130   }
131 
132   @Override
133   public SpatialRelation relate(Shape other) {
134     final SpatialRelation bboxSect = bbox.relate(other);
135     if (bboxSect == SpatialRelation.DISJOINT || bboxSect == SpatialRelation.WITHIN)
136       return bboxSect;
137 
138     final boolean containsWillShortCircuit = (other instanceof Point) ||
139         relateContainsShortCircuits();
140     SpatialRelation sect = null;
141     for (Shape shape : shapes) {
142       SpatialRelation nextSect = shape.relate(other);
143 
144       if (sect == null) {//first pass
145         sect = nextSect;
146       } else {
147         sect = sect.combine(nextSect);
148       }
149 
150       if (sect == INTERSECTS)
151         return INTERSECTS;
152 
153       if (sect == CONTAINS && containsWillShortCircuit)
154         return CONTAINS;
155     }
156     return sect;
157   }
158 
159   /**
160    * Called by relate() to determine whether to return early if it finds
161    * CONTAINS, instead of checking the remaining shapes. It will do so without
162    * calling this method if the "other" shape is a Point.  If a remaining shape
163    * finds INTERSECTS, then INTERSECTS will be returned.  The only problem with
164    * this returning true is that if some of the shapes overlap, it's possible
165    * that the result of relate() could be dependent on the order of the shapes,
166    * which could be unexpected / wrong depending on the application. The default
167    * implementation returns true because it probably doesn't matter.  If it
168    * does, a subclass could add a boolean flag that this method could return.
169    * That flag could be initialized to true only if the shapes are mutually
170    * disjoint.
171    *
172    * @see #computeMutualDisjoint(java.util.List) .
173    */
174   protected boolean relateContainsShortCircuits() {
175     return true;
176   }
177 
178   /**
179    * Computes whether the shapes are mutually disjoint. This is a utility method
180    * offered for use by a subclass implementing {@link #relateContainsShortCircuits()}.
181    * <b>Beware: this is an O(N^2) algorithm.</b>.  Consequently, consider safely
182    * assuming non-disjoint if shapes.size() > 10 or something.  And if all shapes
183    * are a Point then the result of this method doesn't ultimately matter.
184    */
185   protected static boolean computeMutualDisjoint(List<? extends Shape> shapes) {
186     //WARNING: this is an O(n^2) algorithm.
187     //loop through each shape and see if it intersects any shape before it
188     for (int i = 1; i < shapes.size(); i++) {
189       Shape shapeI = shapes.get(i);
190       for (int j = 0; j < i; j++) {
191         Shape shapeJ = shapes.get(j);
192         if (shapeJ.relate(shapeI).intersects())
193           return false;
194       }
195     }
196     return true;
197   }
198 
199   @Override
200   public double getArea(SpatialContext ctx) {
201     double MAX_AREA = bbox.getArea(ctx);
202     double sum = 0;
203     for (Shape geom : shapes) {
204       sum += geom.getArea(ctx);
205       if (sum >= MAX_AREA)
206         return MAX_AREA;
207     }
208 
209     return sum;
210   }
211 
212   @Override
213   public String toString() {
214     StringBuilder buf = new StringBuilder(100);
215     buf.append("ShapeCollection(");
216     int i = 0;
217     for (Shape shape : shapes) {
218       if (i++ > 0)
219         buf.append(", ");
220       buf.append(shape);
221       if (buf.length() > 150) {
222         buf.append(" ...").append(shapes.size());
223         break;
224       }
225     }
226     buf.append(")");
227     return buf.toString();
228   }

230   @Override
231   public boolean equals(Object o) {
232     if (this == o) return true;
233     if (o == null || getClass() != o.getClass()) return false;
234 
235     ShapeCollection that = (ShapeCollection) o;
236 
237     if (!shapes.equals(that.shapes)) return false;
238 
239     return true;
240   }

242   @Override
243   public int hashCode() {
244     return shapes.hashCode();
245   }
