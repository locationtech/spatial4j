18  package com.spatial4j.core.shape;
19  
20  /**
21   * A circle, also known as a point-radius since that is what it is comprised of.
22   */
23  public interface Circle extends Shape {
24  
25    /**
26     * Expert: Resets the state of this shape given the arguments. This is a
27     * performance feature to avoid excessive Shape object allocation as well as
28     * some argument error checking. Mutable shapes is error-prone so use with
29     * care.
30     */
31    void reset(double x, double y, double radiusDEG);
32  
33    /**
34     * The distance from the point's center to its edge, measured in the same
35     * units as x & y (e.g. degrees if WGS84).
36     */
37    double getRadius();
38  
39  }
