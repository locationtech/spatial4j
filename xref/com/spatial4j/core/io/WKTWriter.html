9   package com.spatial4j.core.io;
10  
11  import com.spatial4j.core.shape.Circle;
12  import com.spatial4j.core.shape.Point;
13  import com.spatial4j.core.shape.Rectangle;
14  import com.spatial4j.core.shape.Shape;
15  import com.spatial4j.core.shape.ShapeCollection;
16  import com.spatial4j.core.shape.impl.BufferedLine;
17  import com.spatial4j.core.shape.impl.BufferedLineString;
18  
19  import java.io.IOException;
20  import java.io.Writer;
21  import java.math.RoundingMode;
22  import java.text.NumberFormat;
23  import java.util.Iterator;
24  
25  public class WKTWriter implements ShapeWriter {
26  
27    @Override
28    public String getFormatName() {
29      return ShapeIO.WKT;
30    }
31  
32  
33    protected StringBuilder append(StringBuilder buffer, Point p, NumberFormat nf) {
34      return buffer.append(nf.format(p.getX())).append(' ').append( nf.format(p.getY()));
35    }
36    
37    @Override
38    public String toString(Shape shape) {
39      NumberFormat nf = LegacyShapeWriter.makeNumberFormat(6);
40      if (shape instanceof Point) {
41        StringBuilder buffer = new StringBuilder();
42        return append(buffer.append("POINT ("),(Point)shape,nf).append(")").toString();
43      }
44      if (shape instanceof Rectangle) {
45        NumberFormat nfMIN = nf;
46        NumberFormat nfMAX = LegacyShapeWriter.makeNumberFormat(6);
47  
48        nfMIN.setRoundingMode( RoundingMode.FLOOR );
49        nfMAX.setRoundingMode( RoundingMode.CEILING );
50        
51        Rectangle rect = (Rectangle)shape;
52        return "ENVELOPE (" +
53            // '(' x1 ',' x2 ',' y2 ',' y1 ')'
54          nfMIN.format(rect.getMinX()) + ", " + nfMAX.format(rect.getMaxX()) + ", "+
55          nfMAX.format(rect.getMaxY()) + ", " + nfMIN.format(rect.getMinY()) + ")";
56  //      
57  //      return "POLYGON(( "+
58  //         nf.format(rect.getMinX()) + " " + nf.format(rect.getMinY()) + ", "+
59  //         nf.format(rect.getMinX()) + " " + nf.format(rect.getMaxY()) + ", "+
60  //         nf.format(rect.getMaxX()) + " " + nf.format(rect.getMaxY()) + ", "+
61  //         nf.format(rect.getMaxX()) + " " + nf.format(rect.getMinY()) + ", "+
62  //         nf.format(rect.getMinX()) + " " + nf.format(rect.getMinY()) + "))";
63      }
64      if (shape instanceof Circle) {
65        Circle c = (Circle) shape;
66  
67        StringBuilder str = new StringBuilder();
68        str.append("BUFFER (POINT (")
69          .append(nf.format(c.getCenter().getX())).append(" ")
70          .append(nf.format(c.getCenter().getY()))
71          .append("), ")
72          .append(nf.format(c.getRadius()))
73          .append(")");
74        return str.toString();
75      }
76      if (shape instanceof BufferedLineString) {
77        BufferedLineString line = (BufferedLineString) shape;
78        StringBuilder str = new StringBuilder();
79  
80        double buf = line.getBuf();
81        if (buf > 0d) {
82          str.append("BUFFER (");
83        }
84  
85        str.append("LINESTRING (");
86        Iterator<BufferedLine> iter = line.getSegments().iterator();
87        while(iter.hasNext()) {
88          BufferedLine seg = iter.next();
89          append(str,seg.getA(),nf).append(", ");
90          if(!iter.hasNext()) {
91            append(str,seg.getB(),nf);
92          }
93        }
94        str.append(")");
95  
96        if (buf > 0d) {
97          str.append(", ").append(nf.format(buf)).append(")");
98        }
99        return str.toString();
100     }
101     if(shape instanceof ShapeCollection) {
102       StringBuilder buffer = new StringBuilder();
103       buffer.append("GEOMETRYCOLLECTION (");
104       boolean first = true;
105       for(Shape sub : ((ShapeCollection<? extends Shape>)shape).getShapes()) {
106         if(!first) {
107           buffer.append(",");
108         }
109         buffer.append(toString(sub));
110         first = false;
111       }
112       buffer.append(")");
113       return buffer.toString();
114     }
115     return LegacyShapeWriter.writeShape(shape, nf);
116   }
117   
118   @Override
119   public void write(Writer output, Shape shape) throws IOException {
120     output.append( toString(shape) );
121   }
122 }
