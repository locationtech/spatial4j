9   package org.locationtech.spatial4j.shape;
10  
11  import org.locationtech.spatial4j.context.SpatialContext;
12  
13  import java.util.List;
14  
15  /**
16   * A factory for {@link Shape}s.
17   * Stateless and thread-safe, except for any returned builders.
18   */
19  public interface ShapeFactory {
20  
21    SpatialContext getSpatialContext();
22  
23    /** If true then {@link #normX(double)} will wrap longitudes outside of the standard
24     * geodetic boundary into it. Example: 181 will become -179. */
25    boolean isNormWrapLongitude();
26  
27    // TODO annoying that a ShapeReader must remember to call norm* methods.  Perhaps
28    //  there should be another shapeFactory impl for shape reading?  :-/  Or not.
29  
30    /** Normalize the 'x' dimension. Might reduce precision or wrap it to be within the bounds. This
31     * is called by {@link org.locationtech.spatial4j.io.ShapeReader}s before creating a shape. */
32    double normX(double x);
33  
34    /** @see #normX(double)  */
35    double normY(double y);
36  
37    /** (disclaimer: the Z dimension isn't fully supported)
38     * @see #normX(double) */
39    double normZ(double z);
40  
41    /**
42     * Called to normalize a value that isn't X or Y or Z. X &amp; Y &amp; Z are normalized via
43     * {@link org.locationtech.spatial4j.context.SpatialContext#normX(double)} &amp; normY &amp; normZ. This
44     * is called by a {@link org.locationtech.spatial4j.io.ShapeReader} before creating a shape.
45     */
46    double normDist(double d);
47  
48    /** Ensure fits in the world bounds. It's called by any shape factory method that
49     * gets an 'x' dimension. */
50    void verifyX(double x);
51  
52    /** @see #verifyX(double)  */
53    void verifyY(double y);
54  
55    /** (disclaimer: the Z dimension isn't fully supported)
56     *  @see #verifyX(double)  */
57    void verifyZ(double z);
58  
59    /** Construct a point. */
60    Point pointXY(double x, double y);
61  
62    /** Construct a point of 3 dimensions.  The implementation might ignore unsupported
63     * dimensions like 'z' or throw an error. */
64    Point pointXYZ(double x, double y, double z);
65  
66    /** Construct a rectangle. */
67    Rectangle rect(Point lowerLeft, Point upperRight);
68  
69    /**
70     * Construct a rectangle. If just one longitude is on the dateline (+/- 180) and if
71     * {@link SpatialContext#isGeo()}
72     * then potentially adjust its sign to ensure the rectangle does not cross the
73     * dateline.
74     */
75    Rectangle rect(double minX, double maxX, double minY, double maxY);
76  
77    /** Construct a circle. The units of "distance" should be the same as x &amp; y. */
78    Circle circle(double x, double y, double distance);
79  
80    /** Construct a circle. The units of "distance" should be the same as x &amp; y. */
81    Circle circle(Point point, double distance);
82  
83    /** Constructs a line string with a possible buffer. It's an ordered sequence of connected vertexes,
84     * with a buffer distance along the line in all directions. There
85     * is no official shape/interface for it so we just return Shape. */
86    @Deprecated // use a builder
87    Shape lineString(List<Point> points, double buf);
88  
89    /** Construct a ShapeCollection, analogous to an OGC GeometryCollection. */
90    @Deprecated // use a builder
91    <S extends Shape> ShapeCollection<S> multiShape(List<S> coll);
92  
93    // BUILDERS:
94  
95    /** (Builder) Constructs a line string, with a possible buffer.
96     * It's an ordered sequence of connected vertexes.
97     * There is no official shape/interface for it yet so we just return Shape. */
98    LineStringBuilder lineString();
99  
100   /** (Builder) Constructs a polygon.
101    * There is no official shape/interface for it yet so we just return Shape. */
102   PolygonBuilder polygon();
103 
104   /** (Builder) Constructs a Shape aggregate in which each component/member
105    * is an instance of the specified class.
106    */
107   <T extends Shape> MultiShapeBuilder<T> multiShape(Class<T> shapeClass);
108 
109   /** (Builder) Constructs a MultiPoint. */
110   MultiPointBuilder multiPoint();
111 
112   /** (Builder) Constructs a MultiLineString, or possibly the result of that buffered. */
113   MultiLineStringBuilder multiLineString();
114 
115   /** (Builder) Constructs a MultiPolygon. */
116   MultiPolygonBuilder multiPolygon();
117 
118   // misc:
119   //Shape buffer(Shape shape);  ?
120 
121   // TODO need Polygon shape
122   // TODO need LineString shape
123   // TODO need BufferedLineString shape
124   // TODO need ShapeCollection to be typed
125 
126   /** Builds a point and returns the generic specified type (usually whatever "this" is). */
127   interface PointsBuilder<T> {
128     /** @see ShapeFactory#pointXY(double, double) */
129     T pointXY(double x, double y);
130     /** @see ShapeFactory#pointXYZ(double, double, double) */
131     T pointXYZ(double x, double y, double z);
132   }
133 
134   /** @see #lineString() */
135   interface LineStringBuilder extends PointsBuilder<LineStringBuilder> {
136     // TODO add dimensionality hint method?
137 
138     LineStringBuilder buffer(double distance);
139 
140     Shape build();
141   }
142 
143   /** @see #polygon() */
144   interface PolygonBuilder extends PointsBuilder<PolygonBuilder> {
145     // TODO add dimensionality hint method?
146 
147     /** Starts a new hole. You must add at least 4 points; furthermore the first and last must be the same.
148      * And don't forget to call {@link HoleBuilder#endHole()}! */
149     HoleBuilder hole();
150 
151     /** Builds the polygon and renders this builder instance invalid. */
152     Shape build();// never a Rect
153 
154     Shape buildOrRect();
155 
156     interface HoleBuilder extends PointsBuilder<HoleBuilder> {
157       /** Finishes the hole and returns the {@link PolygonBuilder}.*/
158       PolygonBuilder endHole();
159     }
160   }
161 
162   // TODO add dimensionality hint method to the multi* builders?
163 
164   /** @see #multiShape(Class) */
165   interface MultiShapeBuilder<T extends Shape> {
166     // TODO add dimensionality hint method?
167 
168     MultiShapeBuilder<T> add(T shape);
169 
170     //ShapeCollection<T> build(); TODO wait till it's a typed interface
171     Shape build();
172   }
173 
174   /** @see #multiPoint() */
175   interface MultiPointBuilder extends PointsBuilder<MultiPointBuilder> {
176 
177     Shape build();  // TODO MultiShape<Point>
178   }
179 
180   /** @see #multiLineString() */
181   interface MultiLineStringBuilder {
182 
183     LineStringBuilder lineString();
184 
185     MultiLineStringBuilder add(LineStringBuilder lineStringBuilder);
186 
187     Shape build(); // TODO MultiShape<LineString>
188   }
189 
190   /** @see #multiPolygon() */
191   interface MultiPolygonBuilder {
192 
193     PolygonBuilder polygon();
194 
195     MultiPolygonBuilder add(PolygonBuilder polygonBuilder);
196 
197     Shape build(); // TODO MultiShape<Polygon>
198   }
199 }
