12  import org.locationtech.spatial4j.shape.Rectangle;

14  import java.util.Iterator;
15  import java.util.Map;
16  import java.util.TreeMap;

18  /**
19   * (INTERNAL) Calculates the minimum bounding box given a bunch of rectangles (ranges).  It's a temporary object and not
20   * thread-safe; throw it away when done.
21   * For a cartesian space, the calculations are trivial but it is not for geodetic.  For
22   * geodetic, it must maintain an ordered set of disjoint ranges as each range is provided.
23   */
24  public class BBoxCalculator {
25    
26    private final SpatialContext ctx;
27  
28    private double minY = Double.POSITIVE_INFINITY;
29    private double maxY = Double.NEGATIVE_INFINITY;
30  
31    private double minX = Double.POSITIVE_INFINITY;
32    private double maxX = Double.NEGATIVE_INFINITY;
33  
34    /** Sorted list of <em>disjoint</em> X ranges keyed by maxX and with minX stored as the "value". */
35    private TreeMap<Double, Double> ranges; // maxX -> minX
36  
37    // note: The use of a TreeMap of Double objects is a bit heavy for the points-only use-case.  In such a case,
38    //  we could instead maintain an array of longitudes we just add onto during expandXRange(). A simplified version
39    //  of the processRanges() method could be used that initially sorts and then proceeds in a similar but simplified
40    //  fashion.
41  
42    public BBoxCalculator(SpatialContext ctx) {
43      this.ctx = ctx;
44    }
45  
46    public void expandRange(Rectangle rect) {
47      expandRange(rect.getMinX(), rect.getMaxX(), rect.getMinY(), rect.getMaxY());
48    }
49  
50    public void expandRange(final double minX, final double maxX, double minY, double maxY) {
51      this.minY = Math.min(this.minY, minY);
52      this.maxY = Math.max(this.maxY, maxY);
53  
54      expandXRange(minX, maxX);
55    }//expandRange
56  
57    public void expandXRange(double minX, double maxX) {
58      if (!ctx.isGeo()) {
59        this.minX = Math.min(this.minX, minX);
60        this.maxX = Math.max(this.maxX, maxX);
61        return;
62      }
63  
64      if (doesXWorldWrap())
65        return;
66  
67      if (ranges == null) {
68        ranges = new TreeMap<>();
69        ranges.put(maxX, minX);
70        return;
71      }
72      assert !ranges.isEmpty();
73      //now the hard part!
74  
75      //Get an iterator starting from the first entry that either contains minX or it's to the right of minX
76      Iterator<Map.Entry<Double, Double>> entryIter = ranges.tailMap(minX, true/*inclusive*/).entrySet().iterator();
77      if (!entryIter.hasNext()) {
78        entryIter = ranges.entrySet().iterator();//wrapped across dateline
79      }
80      Map.Entry<Double, Double> entry = entryIter.next();
81  
82      Double entryMin = entry.getValue();
83      Double entryMax = entry.getKey();
84  
85      //See if entry contains maxX
86      if (rangeContains(entryMin, entryMax, maxX)) {
87        // Easy: either minX is also within this entry in which case nothing to do, or it's below in which case
88        // we just need to update the minX of this entry.
89  
90        //See if entry contains minX
91        if (rangeContains(entryMin, entryMax, minX)) {
92  
93          // This entry & the new range together might wrap the world.
94          if ( (minX != entryMin || maxX != entryMax) //ranges not equal
95              && rangeContains(minX, maxX, entryMin) && rangeContains(minX, maxX, entryMax)) {
96            this.minX = -180;
97            this.maxX = +180;
98            ranges = null;
99          }
100         // Done; nothing to do.
101       } else {
102         //Done: Update entry's start to be minX
103         // note:  TreeMap's Map.Entry doesn't support setting the value :-(  So we remove & add the entry.
104         entryIter.remove();
105         ranges.put(entryMax, minX);
106       }
107 
108     } else {//entry does NOT contain maxX:
109 
110       // We're going to insert an entry.  Determine it's min & max.  While finding the max, we'll delete entries
111       //  that overlap with the new entry.
112 
113       // newMinX is basically the lower of minX & entryMin
114       final Double newMinX  = rangeContains(entryMin, entryMax, minX) ? entryMin : minX;
115 
116       Double newMaxX = maxX;
117       //Loop through entries (starting with current) to see if we should remove it.  At the last one, update newMaxX.
118       while (rangeContains(newMinX, newMaxX, entryMin)) {
119         entryIter.remove();//remove entry!
120         if (!rangeContains(minX, maxX, entryMax)) {
121           newMaxX = entryMax;//adjust newMaxX and stop.
122           break;
123         }
124         // get new entry:
125         if (!entryIter.hasNext()) {
126           if (ranges.isEmpty()) {
127             break;
128           }
129           //wrap around (can only happen once)
130           entryIter = ranges.entrySet().iterator();
131         }
132         entry = entryIter.next();
133         entryMin = entry.getValue();
134         entryMax = entry.getKey();
135       }
136 
137       //Add entry
138       ranges.put(newMaxX, newMinX);
139     }
140   }
141 
142   private void processRanges() {
143     if (ranges.size() == 1) { // an optimization
144       Map.Entry<Double, Double> rangeEntry = ranges.firstEntry();
145       minX = rangeEntry.getValue();
146       maxX = rangeEntry.getKey();
147     } else {
148       // Find the biggest gap. Whenever we do, update minX & maxX for the rect opposite of the gap.
149       Map.Entry<Double, Double> prevRange = ranges.lastEntry();
150       double biggestGap = 0;
151       double possibleRemainingGap = 360;  //calculating this enables us to exit early; often on the first lap!
152       for (Map.Entry<Double, Double> range : ranges.entrySet()) {
153         // calc width of this range and the gap before it.
154         double widthPlusGap = range.getKey() - prevRange.getKey();// this max - last max
155         if (widthPlusGap < 0) {
156           widthPlusGap += 360;
157         }
158         double gap = range.getValue() - prevRange.getKey(); // this min - last max
159         if (gap < 0) {
160           gap += 360;
161         }
162         // reduce possibleRemainingGap by this range width and trailing gap.
163         possibleRemainingGap -= widthPlusGap;
164         if (gap > biggestGap) {
165           biggestGap = gap;
166           minX = range.getValue();
167           maxX = prevRange.getKey();
168           if (possibleRemainingGap <= biggestGap) {
169             break;// no point in continuing
170           }
171         }
172         prevRange = range;
173       }
174     }
175     // Null out the ranges to signify we processed them
176     ranges = null;
177   }
178 
179   private static boolean rangeContains(double minX, double maxX, double x) {
180     if (minX <= maxX)
181       return x >= minX && x <= maxX;
182     else
183       return x >= minX || x <= maxX;
184   }
185 
186   public boolean doesXWorldWrap() {
187     assert ctx.isGeo();
188     //note: not dependent on "ranges", since once we expand to world bounds then ranges is null'ed out
189     return minX == -180 && maxX == 180;
190   }
191 
192   public Rectangle getBoundary() {
193     return ctx.makeRectangle(getMinX(), getMaxX(), getMinY(), getMaxY());
194   }
195 
196   public double getMinX() {
197     if (ranges != null) {
198       processRanges();
199     }
200     return minX;
201   }
202 
203   public double getMaxX() {
204     if (ranges != null) {
205       processRanges();
206     }
207     return maxX;
208   }
209 
210   public double getMinY() {
211     return minY;
212   }
213 
214   public double getMaxY() {
215     return maxY;
216   }
