27  /**
28   * A circle, also known as a point-radius, based on a {@link
29   * com.spatial4j.core.distance.DistanceCalculator} which does all the work. This
30   * implementation should work for both cartesian 2D and geodetic sphere
31   * surfaces.
32   */
33  public class CircleImpl implements Circle {
34  
35    protected final SpatialContext ctx;
36  
37    protected final Point point;
38    protected double radiusDEG;
39  
40    // calculated & cached
41    protected Rectangle enclosingBox;
42  
43    public CircleImpl(Point p, double radiusDEG, SpatialContext ctx) {
44      //We assume any validation of params already occurred (including bounding dist)
45      this.ctx = ctx;
46      this.point = p;
47      this.radiusDEG = point.isEmpty() ? Double.NaN : radiusDEG;
48      this.enclosingBox = point.isEmpty() ? ctx.makeRectangle(Double.NaN, Double.NaN, Double.NaN, Double.NaN) :
49        ctx.getDistCalc().calcBoxByDistFromPt(point, this.radiusDEG, ctx, null);
50    }
51  
52    @Override
53    public void reset(double x, double y, double radiusDEG) {
54      assert ! isEmpty();
55      point.reset(x, y);
56      this.radiusDEG = radiusDEG;
57      this.enclosingBox = ctx.getDistCalc().calcBoxByDistFromPt(point, this.radiusDEG, ctx, enclosingBox);
58    }
59  
60    @Override
61    public boolean isEmpty() {
62      return point.isEmpty();
63    }
64  
65    @Override
66    public Point getCenter() {
67      return point;
68    }
69  
70    @Override
71    public double getRadius() {
72      return radiusDEG;
73    }
74  
75    @Override
76    public double getArea(SpatialContext ctx) {
77      if (ctx == null) {
78        return Math.PI * radiusDEG * radiusDEG;
79      } else {
80        return ctx.getDistCalc().area(this);
81      }
82    }
83  
84    @Override
85    public Circle getBuffered(double distance, SpatialContext ctx) {
86      return ctx.makeCircle(point, distance + radiusDEG);
87    }
88  
89    public boolean contains(double x, double y) {
90      return ctx.getDistCalc().within(point, x, y, radiusDEG);
91    }
92  
93    @Override
94    public boolean hasArea() {
95      return radiusDEG > 0;
96    }

103     return enclosingBox;

106   @Override
107   public SpatialRelation relate(Shape other) {
108 //This shortcut was problematic in testing due to distinctions of CONTAINS/WITHIN for no-area shapes (lines, points).
109 //    if (distance == 0) {
110 //      return point.relate(other,ctx).intersects() ? SpatialRelation.WITHIN : SpatialRelation.DISJOINT;
111 //    }
112     if (isEmpty() || other.isEmpty())
113       return SpatialRelation.DISJOINT;
114     if (other instanceof Point) {
115       return relate((Point) other);
116     }
117     if (other instanceof Rectangle) {
118       return relate((Rectangle) other);
119     }
120     if (other instanceof Circle) {
121       return relate((Circle) other);
122     }
123     return other.relate(this).transpose();
124   }

126   public SpatialRelation relate(Point point) {
127     return contains(point.getX(),point.getY()) ? SpatialRelation.CONTAINS : SpatialRelation.DISJOINT;
128   }

130   public SpatialRelation relate(Rectangle r) {
131     //Note: Surprisingly complicated!
132 
133     //--We start by leveraging the fact we have a calculated bbox that is "cheaper" than use of DistanceCalculator.
134     final SpatialRelation bboxSect = enclosingBox.relate(r);
135     if (bboxSect == SpatialRelation.DISJOINT || bboxSect == SpatialRelation.WITHIN)
136       return bboxSect;
137     else if (bboxSect == SpatialRelation.CONTAINS && enclosingBox.equals(r))//nasty identity edge-case
138       return SpatialRelation.WITHIN;
139     //bboxSect is INTERSECTS or CONTAINS
140     //The result can be DISJOINT, CONTAINS, or INTERSECTS (not WITHIN)
141 
142     return relateRectanglePhase2(r, bboxSect);
143   }
144 
145   protected SpatialRelation relateRectanglePhase2(final Rectangle r, SpatialRelation bboxSect) {
146     // DOES NOT WORK WITH GEO CROSSING DATELINE OR WORLD-WRAP. Other methods handle such cases.
147 
148     //At this point, the only thing we are certain of is that circle is *NOT* WITHIN r, since the
149     // bounding box of a circle MUST be within r for the circle to be within r.
150 
151     //Quickly determine if they are DISJOINT or not.
152     // Find the closest & farthest point to the circle within the rectangle
153     final double closestX, farthestX;
154     final double xAxis = getXAxis();
155     if (xAxis < r.getMinX()) {
156       closestX = r.getMinX();
157       farthestX = r.getMaxX();
158     } else if (xAxis > r.getMaxX()) {
159       closestX = r.getMaxX();
160       farthestX = r.getMinX();
161     } else {
162       closestX = xAxis; //we don't really use this value but to check this condition
163       farthestX = r.getMaxX() - xAxis > xAxis - r.getMinX() ? r.getMaxX() : r.getMinX();
164     }
165 
166     final double closestY, farthestY;
167     final double yAxis = getYAxis();
168     if (yAxis < r.getMinY()) {
169       closestY = r.getMinY();
170       farthestY = r.getMaxY();
171     } else if (yAxis > r.getMaxY()) {
172       closestY = r.getMaxY();
173       farthestY = r.getMinY();
174     } else {
175       closestY = yAxis; //we don't really use this value but to check this condition
176       farthestY = r.getMaxY() - yAxis > yAxis - r.getMinY() ? r.getMaxY() : r.getMinY();
177     }
178 
179     //If r doesn't overlap an axis, then could be disjoint. Test closestXY
180     if (xAxis != closestX && yAxis != closestY) {
181       if (!contains(closestX, closestY))
182         return SpatialRelation.DISJOINT;
183     } // else CAN'T be disjoint if spans axis because earlier bbox check ruled that out
184 
185     //Now, we know it's *NOT* DISJOINT and it's *NOT* WITHIN either.
186     // Does circle CONTAINS r or simply intersect it?
187 
188     //If circle contains r, then its bbox MUST also CONTAIN r.
189     if (bboxSect != SpatialRelation.CONTAINS)
190       return SpatialRelation.INTERSECTS;
191 
192     //If the farthest point of r away from the center of the circle is contained, then all of r is
193     // contained.
194     if (!contains(farthestX, farthestY))
195       return SpatialRelation.INTERSECTS;
196 
197     //geodetic detection of farthest Y when rect crosses x axis can't be reliably determined, so
198     // check other corner too, which might actually be farthest
199     if (point.getY() != getYAxis()) {//geodetic
200       if (yAxis == closestY) {//r crosses north to south over x axis (confusing)
201         double otherY = (farthestY == r.getMaxY() ? r.getMinY() : r.getMaxY());
202         if (!contains(farthestX, otherY))
203           return SpatialRelation.INTERSECTS;
204       }
205     }
206    
207     return SpatialRelation.CONTAINS;
208   }

224   public SpatialRelation relate(Circle circle) {
225     double crossDist = ctx.getDistCalc().distance(point, circle.getCenter());
226     double aDist = radiusDEG, bDist = circle.getRadius();
227     if (crossDist > aDist + bDist)
228       return SpatialRelation.DISJOINT;
229     if (crossDist < aDist && crossDist + bDist <= aDist)
230       return SpatialRelation.CONTAINS;
231     if (crossDist < bDist && crossDist + aDist <= bDist)
232       return SpatialRelation.WITHIN;
233 
234     return SpatialRelation.INTERSECTS;
235   }

247   /**
248    * All {@link Circle} implementations should use this definition of {@link Object#equals(Object)}.
249    */
250   public static boolean equals(Circle thiz, Object o) {
251     assert thiz != null;
252     if (thiz == o) return true;
253     if (!(o instanceof Circle)) return false;
254 
255     Circle circle = (Circle) o;
256 
257     if (!thiz.getCenter().equals(circle.getCenter())) return false;
258     if (Double.compare(circle.getRadius(), thiz.getRadius()) != 0) return false;
259 
260     return true;
261   }

268   /**
269    * All {@link Circle} implementations should use this definition of {@link Object#hashCode()}.
270    */
271   public static int hashCode(Circle thiz) {
272     int result;
273     long temp;
274     result = thiz.getCenter().hashCode();
275     temp = thiz.getRadius() != +0.0d ? Double.doubleToLongBits(thiz.getRadius()) : 0L;
276     result = 31 * result + (int) (temp ^ (temp >>> 32));
277     return result;
278   }
