28  /**
29   * Utilities for encoding and decoding <a href="http://en.wikipedia.org/wiki/Geohash">geohashes</a>.
30   * <p>
31   * This class isn't used by any other part of Spatial4j; it's included largely for convenience of
32   * software using Spatial4j. There are other open-source libraries that have more comprehensive
33   * geohash utilities but providing this one avoids an additional dependency for what's a small
34   * amount of code.  <em>If you're using Spatial4j just for this class, consider alternatives.</em>
35   * <p>
36   * This code originally came from <a href="https://issues.apache.org/jira/browse/LUCENE-1512">
37   * Apache Lucene, LUCENE-1512</a>.
38   */
39  public class GeohashUtils {
40  
41    private static final char[] BASE_32 = {'0', '1', '2', '3', '4', '5', '6',
42        '7', '8', '9', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'j', 'k', 'm', 'n',
43        'p', 'q', 'r', 's', 't', 'u', 'v', 'w', 'x', 'y', 'z'};//note: this is sorted
44  
45    private static final int[] BASE_32_IDX;//sparse array of indexes from '0' to 'z'
46  
47    public static final int MAX_PRECISION = 24;//DWS: I forget what level results in needless more precision but it's about this
48    private static final int[] BITS = {16, 8, 4, 2, 1};
49  
50    static {
51      BASE_32_IDX = new int[BASE_32[BASE_32.length-1] - BASE_32[0] + 1];
52      assert BASE_32_IDX.length < 100;//reasonable length
53      Arrays.fill(BASE_32_IDX,-500);
54      for (int i = 0; i < BASE_32.length; i++) {
55        BASE_32_IDX[BASE_32[i] - BASE_32[0]] = i;
56      }
57    }
58  
59    private GeohashUtils() {
60    }
61  
62    /**
63     * Encodes the given latitude and longitude into a geohash
64     *
65     * @param latitude Latitude to encode
66     * @param longitude Longitude to encode
67     * @return Geohash encoding of the longitude and latitude
68     */
69    public static String encodeLatLon(double latitude, double longitude) {
70      return encodeLatLon(latitude, longitude, 12);
71    }
72  
73    public static String encodeLatLon(double latitude, double longitude, int precision) {
74      double[] latInterval = {-90.0, 90.0};
75      double[] lngInterval = {-180.0, 180.0};
76  
77      final StringBuilder geohash = new StringBuilder(precision);
78      boolean isEven = true;
79  
80      int bit = 0;
81      int ch = 0;
82  
83      while (geohash.length() < precision) {
84        double mid = 0.0;
85        if (isEven) {
86          mid = (lngInterval[0] + lngInterval[1]) / 2D;
87          if (longitude > mid) {
88            ch |= BITS[bit];
89            lngInterval[0] = mid;
90          } else {
91            lngInterval[1] = mid;
92          }
93        } else {
94          mid = (latInterval[0] + latInterval[1]) / 2D;
95          if (latitude > mid) {
96            ch |= BITS[bit];
97            latInterval[0] = mid;
98          } else {
99            latInterval[1] = mid;
100         }
101       }
102 
103       isEven = !isEven;
104 
105       if (bit < 4) {
106         bit++;
107       } else {
108         geohash.append(BASE_32[ch]);
109         bit = 0;
110         ch = 0;
111       }
112     }
113 
114     return geohash.toString();
115   }

117   /**
118    * Decodes the given geohash into a latitude and longitude
119    *
120    * @param geohash Geohash to deocde
121    * @return Array with the latitude at index 0, and longitude at index 1
122    */
123   public static Point decode(String geohash, SpatialContext ctx) {
124     Rectangle rect = decodeBoundary(geohash,ctx);
125     double latitude = (rect.getMinY() + rect.getMaxY()) / 2D;
126     double longitude = (rect.getMinX() + rect.getMaxX()) / 2D;
127     return ctx.makePoint(longitude,latitude);
128 	}
129 
130   /** Returns min-max lat, min-max lon. */
131   public static Rectangle decodeBoundary(String geohash, SpatialContext ctx) {
132     double minY = -90, maxY = 90, minX = -180, maxX = 180;
133     boolean isEven = true;
134 
135     for (int i = 0; i < geohash.length(); i++) {
136       char c = geohash.charAt(i);
137       if (c >= 'A' && c <= 'Z')
138         c -= ('A' - 'a');
139       final int cd = BASE_32_IDX[c - BASE_32[0]];//TODO check successful?
140 
141       for (int mask : BITS) {
142         if (isEven) {
143           if ((cd & mask) != 0) {
144             minX = (minX + maxX) / 2D;
145           } else {
146             maxX = (minX + maxX) / 2D;
147           }
148         } else {
149           if ((cd & mask) != 0) {
150             minY = (minY + maxY) / 2D;
151           } else {
152             maxY = (minY + maxY) / 2D;
153           }
154         }
155         isEven = !isEven;
156       }
157 
158     }
159     return ctx.makeRectangle(minX, maxX, minY, maxY);
160   }

162   /** Array of geohashes 1 level below the baseGeohash. Sorted. */
163   public static String[] getSubGeohashes(String baseGeohash) {
164     String[] hashes = new String[BASE_32.length];
165     for (int i = 0; i < BASE_32.length; i++) {//note: already sorted
166       char c = BASE_32[i];
167       hashes[i] = baseGeohash+c;
168     }
169     return hashes;
170   }

172   public static double[] lookupDegreesSizeForHashLen(int hashLen) {
173     return new double[]{hashLenToLatHeight[hashLen], hashLenToLonWidth[hashLen]};
174   }
175 
176   /**
177    * Return the shortest geohash length that will have a width &amp; height &gt;= specified arguments.
178    */
179   public static int lookupHashLenForWidthHeight(double lonErr, double latErr) {
180     //loop through hash length arrays from beginning till we find one.
181     for(int len = 1; len < MAX_PRECISION; len++) {
182       double latHeight = hashLenToLatHeight[len];
183       double lonWidth = hashLenToLonWidth[len];
184       if (latHeight < latErr && lonWidth < lonErr)
185         return len;
186     }
187     return MAX_PRECISION;
188   }
189 
190   /** See the table at http://en.wikipedia.org/wiki/Geohash */
191   private static final double[] hashLenToLatHeight, hashLenToLonWidth;
192   static {
193     hashLenToLatHeight = new double[MAX_PRECISION +1];
194     hashLenToLonWidth = new double[MAX_PRECISION +1];
195     hashLenToLatHeight[0] = 90*2;
196     hashLenToLonWidth[0] = 180*2;
197     boolean even = false;
198     for(int i = 1; i <= MAX_PRECISION; i++) {
199       hashLenToLatHeight[i] = hashLenToLatHeight[i-1]/(even?8:4);
200       hashLenToLonWidth[i] = hashLenToLonWidth[i-1]/(even?4:8);
201       even = ! even;
202     }
203   }
