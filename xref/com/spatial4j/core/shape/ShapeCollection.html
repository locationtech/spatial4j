11  import com.spatial4j.core.context.SpatialContext;
12  import com.spatial4j.core.shape.impl.BBoxCalculator;
13  
14  import java.util.*;
15  
16  import static com.spatial4j.core.shape.SpatialRelation.CONTAINS;
17  import static com.spatial4j.core.shape.SpatialRelation.INTERSECTS;
18  
19  /**
20   * A collection of Shape objects, analogous to an OGC GeometryCollection. The
21   * implementation demands a List (with random access) so that the order can be
22   * retained if an application requires it, although logically it's treated as an
23   * unordered Set, mostly.
24   * <p/>
25   * Ideally, {@link #relate(Shape)} should return the same result no matter what
26   * the shape order is, although the default implementation can be order
27   * dependent when the shapes overlap; see {@link #relateContainsShortCircuits()}.
28   *  To improve performance slightly, the caller could order the shapes by
29   * largest first so that relate() will have a greater chance of
30   * short-circuit'ing sooner.  As the Shape contract states; it may return
31   * intersects when the best answer is actually contains or within. If any shape
32   * intersects the provided shape then that is the answer.
33   * <p/>
34   * This implementation is not optimized for a large number of shapes; relate is
35   * O(N).  A more sophisticated implementation might do an R-Tree based on
36   * bbox'es, for example.
37   */
38  public class ShapeCollection<S extends Shape> extends AbstractList<S> implements Shape {
39  
40    protected final SpatialContext ctx;
41    protected final List<S> shapes;
42    protected final Rectangle bbox;
43  
44    /**
45     * WARNING: {@code shapes} is copied by reference.
46     * @param shapes Copied by reference! (make a defensive copy if caller modifies)
47     * @param ctx
48     */
49    public ShapeCollection(List<S> shapes, SpatialContext ctx) {
50      if (!(shapes instanceof RandomAccess))
51        throw new IllegalArgumentException("Shapes arg must implement RandomAccess: "+shapes.getClass());
52      this.shapes = shapes;
53      this.ctx = ctx;
54      this.bbox = computeBoundingBox(shapes, ctx);
55    }
56  
57    protected Rectangle computeBoundingBox(Collection<? extends Shape> shapes, SpatialContext ctx) {
58      if (shapes.isEmpty())
59        return ctx.makeRectangle(Double.NaN, Double.NaN, Double.NaN, Double.NaN);
60      BBoxCalculator bboxCalc = new BBoxCalculator(ctx);
61      for (Shape geom : shapes) {
62        bboxCalc.expandRange(geom.getBoundingBox());
63      }
64      return bboxCalc.getBoundary();
65    }
66  
67    public List<S> getShapes() {
68      return shapes;
69    }
70  
71    @Override
72    public S get(int index) {
73      return shapes.get(index);
74    }
75  
76    @Override
77    public int size() {
78      return shapes.size();
79    }
80  
81    @Override
82    public Rectangle getBoundingBox() {
83      return bbox;
84    }
85  
86    @Override
87    public Point getCenter() {
88      return bbox.getCenter();
89    }
90  
91    @Override
92    public boolean hasArea() {
93      for (Shape geom : shapes) {
94        if( geom.hasArea() ) {
95          return true;
96        }
97      }
98      return false;
99    }
100 
101   @Override
102   public ShapeCollection getBuffered(double distance, SpatialContext ctx) {
103     List<Shape> bufColl = new ArrayList<Shape>(size());
104     for (Shape shape : shapes) {
105       bufColl.add(shape.getBuffered(distance, ctx));
106     }
107     return ctx.makeCollection(bufColl);
108   }
109 
110   @Override
111   public SpatialRelation relate(Shape other) {
112     final SpatialRelation bboxSect = bbox.relate(other);
113     if (bboxSect == SpatialRelation.DISJOINT || bboxSect == SpatialRelation.WITHIN)
114       return bboxSect;
115 
116     final boolean containsWillShortCircuit = (other instanceof Point) ||
117         relateContainsShortCircuits();
118     SpatialRelation sect = null;
119     for (Shape shape : shapes) {
120       SpatialRelation nextSect = shape.relate(other);
121 
122       if (sect == null) {//first pass
123         sect = nextSect;
124       } else {
125         sect = sect.combine(nextSect);
126       }
127 
128       if (sect == INTERSECTS)
129         return INTERSECTS;
130 
131       if (sect == CONTAINS && containsWillShortCircuit)
132         return CONTAINS;
133     }
134     return sect;
135   }
136 
137   /**
138    * Called by relate() to determine whether to return early if it finds
139    * CONTAINS, instead of checking the remaining shapes. It will do so without
140    * calling this method if the "other" shape is a Point.  If a remaining shape
141    * finds INTERSECTS, then INTERSECTS will be returned.  The only problem with
142    * this returning true is that if some of the shapes overlap, it's possible
143    * that the result of relate() could be dependent on the order of the shapes,
144    * which could be unexpected / wrong depending on the application. The default
145    * implementation returns true because it probably doesn't matter.  If it
146    * does, a subclass could add a boolean flag that this method could return.
147    * That flag could be initialized to true only if the shapes are mutually
148    * disjoint.
149    *
150    * @see #computeMutualDisjoint(java.util.List) .
151    */
152   protected boolean relateContainsShortCircuits() {
153     return true;
154   }
155 
156   /**
157    * Computes whether the shapes are mutually disjoint. This is a utility method
158    * offered for use by a subclass implementing {@link #relateContainsShortCircuits()}.
159    * <b>Beware: this is an O(N^2) algorithm.</b>.  Consequently, consider safely
160    * assuming non-disjoint if shapes.size() > 10 or something.  And if all shapes
161    * are a Point then the result of this method doesn't ultimately matter.
162    */
163   protected static boolean computeMutualDisjoint(List<? extends Shape> shapes) {
164     //WARNING: this is an O(n^2) algorithm.
165     //loop through each shape and see if it intersects any shape before it
166     for (int i = 1; i < shapes.size(); i++) {
167       Shape shapeI = shapes.get(i);
168       for (int j = 0; j < i; j++) {
169         Shape shapeJ = shapes.get(j);
170         if (shapeJ.relate(shapeI).intersects())
171           return false;
172       }
173     }
174     return true;
175   }
176 
177   @Override
178   public double getArea(SpatialContext ctx) {
179     double MAX_AREA = bbox.getArea(ctx);
180     double sum = 0;
181     for (Shape geom : shapes) {
182       sum += geom.getArea(ctx);
183       if (sum >= MAX_AREA)
184         return MAX_AREA;
185     }
186 
187     return sum;
188   }
189 
190   @Override
191   public String toString() {
192     StringBuilder buf = new StringBuilder(100);
193     buf.append("ShapeCollection(");
194     int i = 0;
195     for (Shape shape : shapes) {
196       if (i++ > 0)
197         buf.append(", ");
198       buf.append(shape);
199       if (buf.length() > 150) {
200         buf.append(" ...").append(shapes.size());
201         break;
202       }
203     }
204     buf.append(")");
205     return buf.toString();
206   }

208   @Override
209   public boolean equals(Object o) {
210     if (this == o) return true;
211     if (o == null || getClass() != o.getClass()) return false;
212 
213     ShapeCollection that = (ShapeCollection) o;
214 
215     if (!shapes.equals(that.shapes)) return false;
216 
217     return true;
218   }

221   public int hashCode() {
222     return shapes.hashCode();
223   }

225   @Override
226   public SpatialContext getContext() {
227     return ctx;
228   }
