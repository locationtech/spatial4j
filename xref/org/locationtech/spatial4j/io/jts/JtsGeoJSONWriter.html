11  import java.io.IOException;
12  import java.io.Writer;
13  import java.text.NumberFormat;

31  public class JtsGeoJSONWriter extends GeoJSONWriter {
32  
33    protected final JtsSpatialContext ctx;
34  
35    public JtsGeoJSONWriter(JtsSpatialContext ctx, SpatialContextFactory factory) {
36      super(ctx, factory);
37      this.ctx = ctx;
38    }
39  
40    // --------------------------------------------------------------
41    // Write JTS To GeoJSON
42    // --------------------------------------------------------------
43  
44    protected void write(Writer output, NumberFormat nf, Coordinate coord) throws IOException {
45      output.write('[');
46      output.write(nf.format(coord.x));
47      output.write(',');
48      output.write(nf.format(coord.y));
49      output.write(']');
50    }
51  
52    protected void write(Writer output, NumberFormat nf, CoordinateSequence coordseq)
53        throws IOException {
54      output.write('[');
55      int dim = coordseq.getDimension();
56      for (int i = 0; i < coordseq.size(); i++) {
57        if (i > 0) {
58          output.write(',');
59        }
60        output.write('[');
61        output.write(nf.format(coordseq.getOrdinate(i, 0)));
62        output.write(',');
63        output.write(nf.format(coordseq.getOrdinate(i, 1)));
64        if (dim > 2) {
65          double v = coordseq.getOrdinate(i, 2);
66          if (!Double.isNaN(v)) {
67            output.write(',');
68            output.write(nf.format(v));
69          }
70        }
71        output.write(']');
72      }
73      output.write(']');
74    }
75  
76    protected void write(Writer output, NumberFormat nf, Coordinate[] coord) throws IOException {
77      output.write('[');
78      for (int i = 0; i < coord.length; i++) {
79        if (i > 0) {
80          output.append(',');
81        }
82        write(output, nf, coord[i]);
83      }
84      output.write(']');
85    }
86  
87    protected void write(Writer output, NumberFormat nf, Polygon p) throws IOException {
88      output.write('[');
89      write(output, nf, p.getExteriorRing().getCoordinateSequence());
90      for (int i = 0; i < p.getNumInteriorRing(); i++) {
91        output.append(',');
92        write(output, nf, p.getInteriorRingN(i).getCoordinateSequence());
93      }
94      output.write(']');
95    }
96  
97    public void write(Writer output, Geometry geom) throws IOException {
98      NumberFormat nf = LegacyShapeWriter.makeNumberFormat(6);
99      if (geom instanceof Point) {
100       Point v = (Point) geom;
101       output.append("{\"type\":\"Point\",\"coordinates\":");
102       write(output, nf, v.getCoordinate());
103       output.append("}");
104       return;
105     } else if (geom instanceof Polygon) {
106       output.append("{\"type\":\"Polygon\",\"coordinates\":");
107       write(output, nf, (Polygon) geom);
108       output.append("}");
109       return;
110     } else if (geom instanceof LineString) {
111       LineString v = (LineString) geom;
112       output.append("{\"type\":\"LineString\",\"coordinates\":");
113       write(output, nf, v.getCoordinateSequence());
114       output.append("}");
115       return;
116     } else if (geom instanceof MultiPoint) {
117       MultiPoint v = (MultiPoint) geom;
118       output.append("{\"type\":\"MultiPoint\",\"coordinates\":");
119       write(output, nf, v.getCoordinates());
120       output.append("}");
121       return;
122     } else if (geom instanceof MultiLineString) {
123       MultiLineString v = (MultiLineString) geom;
124       output.append("{\"type\":\"MultiLineString\",\"coordinates\":[");
125       for (int i = 0; i < v.getNumGeometries(); i++) {
126         if (i > 0) {
127           output.append(',');
128         }
129         write(output, nf, v.getGeometryN(i).getCoordinates());
130       }
131       output.append("]}");
132     } else if (geom instanceof MultiPolygon) {
133       MultiPolygon v = (MultiPolygon) geom;
134       output.append("{\"type\":\"MultiPolygon\",\"coordinates\":[");
135       for (int i = 0; i < v.getNumGeometries(); i++) {
136         if (i > 0) {
137           output.append(',');
138         }
139         write(output, nf, (Polygon) v.getGeometryN(i));
140       }
141       output.append("]}");
142     } else if (geom instanceof GeometryCollection) {
143       GeometryCollection v = (GeometryCollection) geom;
144       output.append("{\"type\":\"GeometryCollection\",\"geometries\":[");
145       for (int i = 0; i < v.getNumGeometries(); i++) {
146         if (i > 0) {
147           output.append(',');
148         }
149         write(output, v.getGeometryN(i));
150       }
151       output.append("]}");
152     } else {
153       throw new UnsupportedOperationException("unknown: " + geom);
154     }
155   }
156 
157   @Override
158   public void write(Writer output, Shape shape) throws IOException {
159     if (shape == null) {
160       throw new NullPointerException("Shape can not be null");
161     }
162     if (shape instanceof JtsGeometry) {
163       write(output, ((JtsGeometry) shape).getGeom());
164       return;
165     }
166     super.write(output, shape);
167   }
168 }
